00900 void MPCylinderScatterDevice::scatter(size_t this_moment) {
00901    // outer loop: frames
00902    // inner loop: block of moments
00903    
00904    using namespace boost::numeric::ublas::detail;
00905 //   cerr << "this_subvector " << this_subvector << endl;
00906 //   sleep(3);
00907    std::vector<double>& sfs = scatterfactors.get_all();
00908 
00909    DivAssignment zeronode_assignment(partitioncomm_.size(),0,NF);
00910    size_t NMAXF = zeronode_assignment.max();
00911    size_t NNPP = partitioncomm_.size();

00913    if (NNPP==1) {
00914        size_t NTHREADS = worker_threads.size();
00915        offset = (this_moment%NTHREADS)*NF;
00916    } else {
00917        offset = (this_moment%NNPP)*NMAXF;       
00918    }
00919 
00920    DivAssignment assignment(partitioncomm_.size(),partitioncomm_.rank(),NF);
00921    size_t NMYF = assignment.size();
00922    fftw_complex* p_a = &(at_[offset]);
00923    
00924    std::pair<long,long> moment = multipole_index_[this_moment];
00925 
00926    CartesianCoor3D o = Params::Inst()->scattering.average.orientation.axis;
00927     
00928     // constructs a base out of thin air
00929    CartesianVectorBase base(o);
00930    CartesianCoor3D qprojected = base.project(qvector_);
00931    CylinderCoor3D qcylinder(qprojected);
00932    double qr = qcylinder.r;
00933    double qphi = qcylinder.phi;
00934    double qz = qcylinder.z;
00935 
00936    long l = moment.first;
00937    long m = moment.second;
00938    
00939    
00940    for(size_t fi = 0; fi < NMYF; ++fi)
00941    {
00942        coor_t* p_data = &(p_coordinates[fi*NA*3]);
00943 
00944        complex<double> A = 0;
00945        for(size_t j = 0; j < NA; ++j) {
00946            double r = p_data[3*j];
00947            double phi = p_data[3*j+1];
00948            double z = p_data[3*j+2];
00949        
00950            double esf = sfs[j];
00951            
00952            double parallel_sign = 1.0;
00953            if ((z!=0) && (qz!=0)) {
00954                parallel_sign = (z*qz) / (abs(z)*abs(qz));           
00955            }
00956 
00957            complex<double> expi = exp(complex<double>(0,parallel_sign*z*qz));
00958            double p = r*qr;
00959            double psiphi = phi-qphi; // review this!
00960            
00961            if (m==0 && l==0) {
00962                A += expi * (double)boost::math::cyl_bessel_j(0,p) * esf ;
00963            } else if (m==0) {              
00964                // if func==1
00965                complex<double> fac1 = 2.0*powf(-1.0,l)*boost::math::cyl_bessel_j(2*l,p);
00966                A += sqrt(0.5)*fac1 * expi *cos(2*l*psiphi) * esf;               
00967            } else if (m==1) {
00968                // if func==2
00969                complex<double> fac1 = 2.0*powf(-1.0,l)*boost::math::cyl_bessel_j(2*l,p);
00970                A += sqrt(0.5)*fac1 * expi *sin(2*l*psiphi) * esf;          
00971            } else if (m==2) {
00972                // if func==3
00973                complex<double> fac2 = complex<double>(0,1.0)*double(2.0*powf(-1.0,l-1)*boost::math::cyl_bessel_j(2*l-1,p));
00974                A += sqrt(0.5)*fac2 * expi *cos((2*l-1)*psiphi) * esf;    
00975            } else if (m==3) {
00976                // if func==4
00977                complex<double> fac2 = complex<double>(0,1.0)*double(2.0*powf(-1.0,l-1)*boost::math::cyl_bessel_j(2*l-1,p));
00978                A += sqrt(0.5)*fac2 * expi *sin((2*l-1)*psiphi) * esf;
00979            }
00980        }
00981        // normalize with integral over phi
00982        double norm = sqrt(2*M_PI);
00983        p_a[fi][0] = norm*A.real();
00984        p_a[fi][1] = norm*A.imag();
